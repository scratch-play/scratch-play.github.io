const STUDIO_ID = '51676986'; const PAGE_LIMIT = 40; // Используем бесплатный прокси для обхода CORS const PROXY_URL = 'https://corsproxy.io?'; let offset = 0; let projectCollection = []; async function fetchStudioProjects() { const statusLabel = document.getElementById('status'); try { // Правильный URL API Scratch const apiTarget = `https://mit.edu{STUDIO_ID}/projects?limit=${PAGE_LIMIT}&offset=${offset}`; const response = await fetch(PROXY_URL + encodeURIComponent(apiTarget)); if (!response.ok) throw new Error(); const data = await response.json(); if (data && data.length > 0) { projectCollection = projectCollection.concat(data); offset += PAGE_LIMIT; statusLabel.textContent = `Загрузка... Получено: ${projectCollection.length}`; setTimeout(fetchStudioProjects, 250); } else { statusLabel.textContent = `Успешно загружено: ${projectCollection.length}`; document.getElementById('search-input').disabled = false; document.getElementById('random-btn').style.display = 'inline-block'; displayProjects(projectCollection); } } catch (err) { statusLabel.style.background = '#ffeef0'; statusLabel.style.color = '#d9383a'; statusLabel.textContent = 'Ошибка загрузки. Попробуйте позже.'; console.error(err); } } function displayProjects(list) { const container = document.getElementById('projects-container'); container.innerHTML = ''; list.forEach(item => { const card = document.createElement('div'); card.className = 'project-card'; // Исправленный путь к иконке проекта const imgUrl = `https://mit.edu{item.id}.png`; card.innerHTML = `
${item.title}
`; card.onclick = () => { toggleView(true); document.getElementById('game-title').textContent = item.title; // Исправленный URL плеера document.getElementById('scratch-player').src = `https://mit.edu{item.id}/embed`; }; container.appendChild(card); }); } // Остальные функции (toggleView, playRandomGame) остаются без изменений
